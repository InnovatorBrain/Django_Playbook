# Complex Lookups using Q objects:
from django.db.models import Q
- queryset = Product.objects.filter(Q(inventory__lt=10) | Q(unit_price__lt=20))
- Product.objects.filter(Q(inventory__lt=10) & ~Q(unit_price__lt=20))
